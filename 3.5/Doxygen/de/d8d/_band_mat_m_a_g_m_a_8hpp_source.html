74template<sp_d T> int BandMatMAGMA<T>::direct_solve(Mat<T>& X, Mat<T>&& B) {
75 suanpan_assert([&] { if(this->n_rows != this->n_cols) throw invalid_argument("requires a square matrix"); });
76
77 auto INFO = 0;
78
79 const auto N = static_cast<magma_int_t>(this->n_rows);
80 const auto KL = static_cast<magma_int_t>(this->l_band);
81 const auto KU = static_cast<magma_int_t>(this->u_band);
82 const auto NRHS = static_cast<magma_int_t>(B.n_cols);
83 const auto LDAB = static_cast<magma_int_t>(this->m_rows);
84 const auto LDB = static_cast<magma_int_t>(B.n_rows);
85
86 const auto LDDAB = magma_roundup(LDAB, 32);
87 const auto LDDB = magma_roundup(LDB, 32);
88 const auto SIZEA = LDDAB * N;
89 const auto SIZEB = LDDB * NRHS;
90
91 release();
92
93 magma_imalloc(&IPIV, N);
94
95 if constexpr(std::is_same_v<T, float>) {
96 using E = float;
97
98 magma_smalloc((magmaFloat_ptr*)&APTR, SIZEA);
99 magma_smalloc((magmaFloat_ptr*)&BPTR, SIZEB);
100
101 magma_ssetmatrix(LDAB, N, (E*)this->memptr(), LDAB, (magmaFloat_ptr)APTR, LDDAB, queue);
102 magma_ssetmatrix(N, NRHS, (E*)B.memptr(), LDB, (magmaFloat_ptr)BPTR, LDDB, queue);
103 magma_sgbsv_native(N, KL, KU, NRHS, (magmaFloat_ptr)APTR, LDDAB, IPIV, (magmaFloat_ptr)BPTR, LDDB, &INFO);
104 magma_sgetmatrix(N, NRHS, (magmaFloat_ptr)BPTR, LDDB, (E*)B.memptr(), LDB, queue);
105
106 X = std::move(B);
107 }
108 else if(Precision::FULL == this->setting.precision) {
109 using E = double;
110
111 magma_dmalloc((magmaDouble_ptr*)&APTR, SIZEA);
112 magma_dmalloc((magmaDouble_ptr*)&BPTR, SIZEB);
113
114 magma_dsetmatrix(LDAB, N, (E*)this->memptr(), LDAB, (magmaDouble_ptr)APTR, LDDAB, queue);
115 magma_dsetmatrix(N, NRHS, (E*)B.memptr(), LDB, (magmaDouble_ptr)BPTR, LDDB, queue);
116 magma_dgbsv_native(N, KL, KU, NRHS, (magmaDouble_ptr)APTR, LDDAB, IPIV, (magmaDouble_ptr)BPTR, LDDB, &INFO);
117 magma_dgetmatrix(N, NRHS, (magmaDouble_ptr)BPTR, LDDB, (E*)B.memptr(), LDB, queue);
118
119 X = std::move(B);
120 }
121 else {
122 magma_smalloc((magmaFloat_ptr*)&APTR, SIZEA);
123 magma_smalloc((magmaFloat_ptr*)&BPTR, SIZEB);
124
125 this->s_memory = this->to_float();
126
127 this->mixed_trs(X, std::forward<Mat<T>>(B), [&](fmat& residual) {
128 // todo: avoid copy
129 magma_ssetmatrix(LDAB, N, this->s_memory.memptr(), LDAB, (magmaFloat_ptr)APTR, LDDAB, queue);
130 magma_ssetmatrix(N, NRHS, residual.memptr(), LDB, (magmaFloat_ptr)BPTR, LDDB, queue);
131 magma_sgbsv_native(N, KL, KU, NRHS, (magmaFloat_ptr)APTR, LDDAB, IPIV, (magmaFloat_ptr)BPTR, LDDB, &INFO);
132 magma_sgetmatrix(N, NRHS, (magmaFloat_ptr)BPTR, LDDB, residual.memptr(), LDB, queue);
133
134 return INFO;
135 });
136 }
137
138 if(0 != INFO)
139 suanpan_error("Error code {} received, the matrix is probably singular.\n", INFO);
140
141 this->pivot.zeros(N);
142 magma_igetmatrix(N, 1, IPIV, N, this->pivot.memptr(), N, queue);
143
144 release();
145
146 return INFO;
147}
