37template<typename T> std::vector<std::vector<T>> sort_color_wp(const suanpan::graph<T>& node_register) {
38 wall_clock timer;
39 timer.tic();
40
41 const auto num_node = node_register.size();
42
43 uvec weight(num_node, fill::none);
44
45 suanpan::for_each(node_register.size(), [&](const size_t I) { weight(I) = node_register[I].size(); });
46
47 auto comparator = [&](const T A, const T B) { return weight[A] > weight[B]; };
48
49 std::multiset<T, decltype(comparator)> degree(comparator);
50 for(T I = 0; I < num_node; ++I) degree.insert(I);
51
52 std::vector<std::vector<T>> color_map;
53
54 while(!degree.empty()) {
55 color_map.emplace_back();
56 auto& color = color_map.back();
57
58 color.emplace_back(*degree.begin());
59 for(auto node = degree.erase(degree.begin()); node != degree.end();) {
60 auto flag = false;
61 for(const auto colored : color) {
62 if(colored == *node) continue;
63 if(node_register[colored].contains(*node)) {
64 flag = true;
65 break;
66 }
67 }
68 if(flag) ++node;
69 else {
70 color.emplace_back(*node);
71 node = degree.erase(node);
72 }
73 }
74 }
75
76 for(auto& color : color_map) color.shrink_to_fit();
77 color_map.shrink_to_fit();
78
79 suanpan_debug("Coloring algorithm takes {:.5E} seconds.\n", timer.toc());
80
81 return color_map;
82}

84template<typename T> std::vector<std::vector<T>> sort_color_mis(const suanpan::graph<T>& node_register) {
85 wall_clock timer;
86 timer.tic();
87
88 const auto num_node = node_register.size();
89
90 uvec weight(num_node, fill::none);
91
92 suanpan::for_each(node_register.size(), [&](const size_t I) { weight(I) = node_register[I].size(); });
93
94 uword counter = num_node;
95 for(const auto I : sort_index(weight, "descend").eval()) weight[I] = --counter;
96
97 auto comparator = [&](const T A, const T B) { return weight[A] > weight[B]; };
98
99 std::multiset<T, decltype(comparator)> degree(comparator);
100 for(T I = 0; I < num_node; ++I) degree.insert(I);
101
102 std::vector<std::vector<T>> color_map;
103
104 while(!degree.empty()) {
105 color_map.emplace_back();
106 auto& color = color_map.back();
107
108 auto degree_copy = degree;
109 do {
110 const auto node_to_color = *degree_copy.begin();
111 color.emplace_back(node_to_color);
112 degree.erase(node_to_color);
113 for(auto neighbour : node_register[node_to_color]) degree_copy.erase(neighbour);
114 }
115 while(!degree_copy.empty());
116 }
117
118 for(auto& color : color_map) color.shrink_to_fit();
119 color_map.shrink_to_fit();
120
121 suanpan_debug("Coloring algorithm takes {:.5E} seconds.\n", timer.toc());
122
123 return color_map;
124}
