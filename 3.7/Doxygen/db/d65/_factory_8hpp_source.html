776template<sp_d T> int Factory<T>::initialize() {
777 reference_dof.clear(); // clear reference dof vector in every step
778
779 if(initialized || n_size == 0) return 0;
780
781 ninja.zeros(n_size);
782 sushi.zeros(n_size);
783
784 reset();
785
786 switch(analysis_type) {
787 case AnalysisType::DISP:
788 initialize_displacement();
789 break;
790 case AnalysisType::EIGEN:
791 initialize_mass();
792 initialize_stiffness();
793 initialize_eigen();
794 break;
795 case AnalysisType::BUCKLE:
796 case AnalysisType::STATICS:
797 initialize_load();
798 initialize_resistance();
799 initialize_displacement();
800 initialize_stiffness();
801 initialize_geometry();
802 break;
803 case AnalysisType::DYNAMICS:
804 initialize_load();
805 initialize_resistance();
806 initialize_damping_force();
807 initialize_nonviscous_force();
808 initialize_inertial_force();
809 initialize_displacement();
810 initialize_velocity();
811 initialize_acceleration();
812 initialize_mass();
813 initialize_damping();
814 initialize_nonviscous();
815 initialize_stiffness();
816 initialize_geometry();
817 break;
818 case AnalysisType::NONE:
819 break;
820 }
821
822 initialize_auxiliary_resistance();
823
824 access::rw(initialized) = true;
825
826 return 0;
827}

1031template<sp_d T> void Factory<T>::commit_energy() {
1032 auto se = std::async([&] { if(!trial_resistance.empty() && !incre_displacement.empty()) strain_energy += .5 * dot(trial_resistance + current_resistance, incre_displacement); });
1033 auto ke = std::async([&] { if(!trial_inertial_force.empty() && !trial_velocity.empty()) kinetic_energy = .5 * dot(global_mass * trial_velocity, trial_velocity); });
1034 auto ve = std::async([&] { if(!trial_damping_force.empty() && !incre_displacement.empty()) viscous_energy += .5 * dot(trial_damping_force + current_damping_force, incre_displacement); });
1035 auto ne = std::async([&] { if(!trial_nonviscous_force.empty() && !incre_displacement.empty()) nonviscous_energy += .5 * dot(trial_nonviscous_force + current_nonviscous_force, incre_displacement); });
1036 auto ce = std::async([&] { if(!trial_displacement.empty() && !incre_resistance.empty()) complementary_energy += .5 * dot(trial_displacement + current_displacement, incre_resistance); });
1037 auto mm = std::async([&] { if(!trial_inertial_force.empty() && !trial_velocity.empty()) momentum = global_mass * trial_velocity; });
1038
1039 se.get();
1040 ke.get();
1041 ve.get();
1042 ne.get();
1043 ce.get();
1044 mm.get();
1045}
