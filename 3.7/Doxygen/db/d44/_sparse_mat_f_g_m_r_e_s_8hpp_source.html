65template<sp_d T> int SparseMatFGMRES<T>::direct_solve(Mat<T>& X, const Mat<T>& B) {
66 const auto N = static_cast<MKL_INT>(B.n_rows);
67 // ReSharper disable once CppRedundantCastExpression
68 const auto R = std::min(static_cast<MKL_INT>(150), N);
69
70 work.zeros((2 * R + 1) * N + R * (R + 9) / 2 + 1);
71
72 X = B;
73
74 const auto precond = this->triplet_mat.diag();
75
76 csr_form<T, int> csr_mat(this->triplet_mat);
77
78 MKL_INT info;
79
80 dfgmres_init(&N, nullptr, nullptr, &info, ipar, dpar, work.memptr());
81 if(0 != info) return info;
82
83 ipar[8] = 1;
84 ipar[9] = 0;
85 ipar[10] = 1; // use preconditioner
86 ipar[11] = 1;
87 dpar[0] = this->setting.tolerance;
88
89 for(auto I = 0llu; I < B.n_cols; ++I) {
90 while(true) {
91 dfgmres(&N, (double*)X.colptr(I), (double*)B.colptr(I), &info, ipar, dpar, work.memptr()); // NOLINT(clang-diagnostic-cast-qual)
92 if(-1 == info || -10 == info || -11 == info || -12 == info) {
93 suanpan_error("Error code {} received.\n", info);
94 return -1;
95 }
96 if(0 == info || 4 == info && dpar[6] <= dpar[0]) {
97 MKL_INT counter;
98 dfgmres_get(&N, (double*)X.colptr(I), (double*)B.colptr(I), &info, ipar, dpar, work.memptr(), &counter); // NOLINT(clang-diagnostic-cast-qual)
99 suanpan_debug("Converged in {} iterations.\n", counter);
100 break;
101 }
102 if(1 == info) {
103 const vec xn(&work[ipar[21] - 1], N);
104 // ReSharper disable once CppInitializedValueIsAlwaysRewritten
105 // ReSharper disable once CppEntityAssignedButNoRead
106 vec yn(&work[ipar[22] - 1], N, false, true);
107 // ReSharper disable once CppDFAUnusedValue
108 yn = csr_mat * xn;
109 }
110 else if(3 == info) {
111 const vec xn(&work[ipar[21] - 1], N);
112 // ReSharper disable once CppInitializedValueIsAlwaysRewritten
113 // ReSharper disable once CppEntityAssignedButNoRead
114 vec yn(&work[ipar[22] - 1], N, false, true);
115 // ReSharper disable once CppDFAUnusedValue
116 yn = xn / precond;
117 }
118 }
119 }
120
121 return info;
122}
