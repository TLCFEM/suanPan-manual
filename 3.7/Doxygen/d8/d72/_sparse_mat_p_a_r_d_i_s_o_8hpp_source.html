107template<sp_d T, la_it MT> int SparseMatBasePARDISO<T, MT>::direct_solve(Mat<T>& X, const Mat<T>& B) {
108 if(!this->factored) {

110 suanpan_error("Error code {} received.\n", info);
111 return SUANPAN_FAIL;
112 }
113 this->factored = true;
114 }
115
116 X.set_size(B.n_rows, B.n_cols);
117
118 const la_it nrhs{static_cast<la_it>(B.n_cols)};
119
120 la_it info{-1};
121 pardiso(pt, &maxfct, &mnum, &mtype, &PARDISO_SOLVE, &csr_mat.n_rows, csr_mat.val_mem(), csr_mat.row_mem(), csr_mat.col_mem(), nullptr, &nrhs, iparm, &msglvl, (void*)B.memptr(), X.memptr(), &info);
122
123 if(0 != info) {
124 suanpan_error("Error code {} received.\n", info);
125 return SUANPAN_FAIL;
126 }
127
128 return SUANPAN_SUCCESS;
129}
