49 template<bool convert, typename ST,std::invocable<ST*> F, typename COO> int alloc(COO& triplet, ST& mumps_job, F& mumps_c) {
50 if(this->factored) return 0;
51
52 dealloc(mumps_job, mumps_c);
53
54 this->factored = true;
55
56 mumps_job.job = -1;
57 mumps_c(&mumps_job);
58
59 triplet.csc_condense();
60
61 if constexpr(convert) {
62 l_irn.set_size(triplet.n_elem);
63 l_jrn.set_size(triplet.n_elem);
64
65 suanpan_for(0, static_cast<int>(triplet.n_elem), [&](const int I) {
66 l_irn[I] = static_cast<int>(triplet.row(I) + 1);
67 l_jrn[I] = static_cast<int>(triplet.col(I) + 1);
68 });
69
70 mumps_job.irn = l_irn.memptr();
71 mumps_job.jcn = l_jrn.memptr();
72 }
73 else {
74 mumps_job.irn = triplet.row_mem();
75 mumps_job.jcn = triplet.col_mem();
76 }
77
78 mumps_job.a = triplet.val_mem();
79
80 mumps_job.n = static_cast<int>(triplet.n_rows);
81 mumps_job.nnz = static_cast<int64_t>(triplet.n_elem);
82
83 mumps_job.icntl[0] = -1;
84 mumps_job.icntl[1] = -1;
85 mumps_job.icntl[2] = -1;
86 mumps_job.icntl[3] = 0;
87 mumps_job.icntl[9] = -2;
88 mumps_job.icntl[13] = 100;
89 mumps_job.icntl[19] = 0; // dense rhs
90 mumps_job.icntl[32] = 1; // determinant
91 mumps_job.icntl[34] = 1; // BLR
92
93 mumps_job.job = 4;
94 mumps_c(&mumps_job);
95
96 if(0 != mumps_job.info[0])
97 suanpan_error("Error code {} received.\n", mumps_job.info[0]);
98
99 return mumps_job.info[0];
100 }

116 int direct_solve(Mat<T>& X, Mat<T>&& B) override {
117 int INFO;
118
119 auto mat_ptr = 0 == this->sym ? &this->triplet_mat : &h_mat;
120
121 if constexpr(std::is_same_v<T, float>) {
122 if(0 != (INFO = alloc<true>(*mat_ptr, smumps_job, smumps_c))) return INFO;
123
124 smumps_job.rhs = B.memptr();
125 smumps_job.lrhs = static_cast<int>(B.n_rows);
126 smumps_job.nrhs = static_cast<int>(B.n_cols);
127
128 run(smumps_job, smumps_c);
129
130 X = std::move(B);
131
132 INFO = smumps_job.info[0];
133 }
134 else if(Precision::FULL == this->setting.precision) {
135 if(0 != (INFO = alloc<true>(*mat_ptr, dmumps_job, dmumps_c))) return INFO;
136
137 dmumps_job.rhs = B.memptr();
138 dmumps_job.lrhs = static_cast<int>(B.n_rows);
139 dmumps_job.nrhs = static_cast<int>(B.n_cols);
140
141 run(dmumps_job, dmumps_c);
142
143 X = std::move(B);
144
145 INFO = dmumps_job.info[0];
146 }
147 else {
148 if(!this->factored) s_mat = triplet_form<float, int>(*mat_ptr, SparseBase::ONE, false);
149
150 if(0 != (INFO = alloc<false>(s_mat, smumps_job, smumps_c))) return INFO;
151
152 INFO = this->mixed_trs(X, std::forward<Mat<T>>(B), [&](fmat& residual) {
153 smumps_job.rhs = residual.memptr();
154 smumps_job.lrhs = static_cast<int>(residual.n_rows);
155 smumps_job.nrhs = static_cast<int>(residual.n_cols);
156
157 run(smumps_job, smumps_c);
158
159 return smumps_job.info[0];
160 });
161 }
162
163 return INFO;
164 }
