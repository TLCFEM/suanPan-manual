50template<sp_d T> int SparseMatFGMRES<T>::direct_solve(Mat<T>& X, const Mat<T>& B) {
51 X.zeros(B.n_rows, B.n_cols);
52
53 csr_form<T, la_it> csr_mat(this->triplet_mat);
54
55 const Col<T> precond = this->triplet_mat.diag() + std::numeric_limits<T>::epsilon();
56
57 Col<int> info(B.n_cols);
58
59 for(auto I = 0llu; I < B.n_cols; ++I) info[I] = fgmres_solve(csr_mat, precond, X.colptr(I), (double*)B.colptr(I), this->setting.tolerance);
60
61 return info.min();
62}
