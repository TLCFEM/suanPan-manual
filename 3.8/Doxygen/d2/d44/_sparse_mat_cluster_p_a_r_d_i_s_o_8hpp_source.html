56template<sp_d T, ezp::matrix_type mtype> int SparseMatBaseClusterPARDISO<T, mtype>::solve_full(Mat<T>& X) {
57 la_it info{-1};
58
59 if(this->factored) info = solver.solve({X.n_rows, X.n_cols, X.memptr()});
60 else {
61 this->factored = true;
62
63 solver.iparm_default_value(1);
64 solver.iparm_reducing_ordering(2);
65 solver.iparm_iterative_refinement(2);
66 solver.iparm_pivoting_perturbation(std::is_same_v<T, double> ? 14 : 7);
67 solver.iparm_weighted_matching(1);
68 ezp::pardiso_set(this->setting.option, solver);
69
70 if(ezp::matrix_type::real_and_nonsymmetric == mtype) {
71 csr_mat = csr_form<T, la_it>(this->triplet_mat, SparseBase::ONE, true);
72 solver.iparm_scaling(1);
73 }
74 else csr_mat = csr_form<T, la_it>(this->triplet_mat.upper(), SparseBase::ONE, true);
75
76 info = solver.solve({csr_mat.n_rows, csr_mat.n_elem, csr_mat.row_mem(), csr_mat.col_mem(), csr_mat.val_mem()}, {X.n_rows, X.n_cols, X.memptr()});
77 }
78
79 if(0 == info) bcast_from_root(X);
80 else suanpan_error("Error code {} received.\n", info);
81
82 return info;
83}
