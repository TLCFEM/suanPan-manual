70template<sp_d T> int BandMatMAGMA<T>::direct_solve(Mat<T>& X, Mat<T>&& B) {
71 suanpan_assert([&] { if(this->n_rows != this->n_cols) throw std::invalid_argument("requires a square matrix"); });
72
73 auto INFO = 0;
74
75 const auto N = static_cast<magma_int_t>(this->n_rows);
76 const auto KL = static_cast<magma_int_t>(this->l_band);
77 const auto KU = static_cast<magma_int_t>(this->u_band);
78 const auto NRHS = static_cast<magma_int_t>(B.n_cols);
79 const auto LDAB = static_cast<magma_int_t>(this->m_rows);
80 const auto LDB = static_cast<magma_int_t>(B.n_rows);
81
82 const auto LDDAB = magma_roundup(LDAB, 32);
83 const auto LDDB = magma_roundup(LDB, 32);
84 const auto SIZEA = LDDAB * N;
85 const auto SIZEB = LDDB * NRHS;
86
87 magma_imalloc(&IPIV, N);
88
89 if constexpr(std::is_same_v<T, float>) {
90 using E = float;
91
92 magma_smalloc((magmaFloat_ptr*)&APTR, SIZEA);
93 magma_smalloc((magmaFloat_ptr*)&BPTR, SIZEB);
94
95 magma_ssetmatrix(LDAB, N, (E*)this->memptr(), LDAB, (magmaFloat_ptr)APTR, LDDAB, queue);
96 magma_ssetmatrix(N, NRHS, (E*)B.memptr(), LDB, (magmaFloat_ptr)BPTR, LDDB, queue);
97 magma_sgbsv_native(N, KL, KU, NRHS, (magmaFloat_ptr)APTR, LDDAB, IPIV, (magmaFloat_ptr)BPTR, LDDB, &INFO);
98 magma_sgetmatrix(N, NRHS, (magmaFloat_ptr)BPTR, LDDB, (E*)B.memptr(), LDB, queue);
99
100 X = std::move(B);
101 }
102 else if(Precision::FULL == this->setting.precision) {
103 using E = double;
104
105 magma_dmalloc((magmaDouble_ptr*)&APTR, SIZEA);
106 magma_dmalloc((magmaDouble_ptr*)&BPTR, SIZEB);
107
108 magma_dsetmatrix(LDAB, N, (E*)this->memptr(), LDAB, (magmaDouble_ptr)APTR, LDDAB, queue);
109 magma_dsetmatrix(N, NRHS, (E*)B.memptr(), LDB, (magmaDouble_ptr)BPTR, LDDB, queue);
110 magma_dgbsv_native(N, KL, KU, NRHS, (magmaDouble_ptr)APTR, LDDAB, IPIV, (magmaDouble_ptr)BPTR, LDDB, &INFO);
111 magma_dgetmatrix(N, NRHS, (magmaDouble_ptr)BPTR, LDDB, (E*)B.memptr(), LDB, queue);
112
113 X = std::move(B);
114 }
115 else {
116 magma_smalloc((magmaFloat_ptr*)&APTR, SIZEA);
117 magma_smalloc((magmaFloat_ptr*)&BPTR, SIZEB);
118
119 this->s_memory = this->to_float();
120
121 this->mixed_trs(X, std::move(B), [&](fmat& residual) {
122 // todo: avoid copy
123 magma_ssetmatrix(LDAB, N, this->s_memory.memptr(), LDAB, (magmaFloat_ptr)APTR, LDDAB, queue);
124 magma_ssetmatrix(N, NRHS, residual.memptr(), LDB, (magmaFloat_ptr)BPTR, LDDB, queue);
125 magma_sgbsv_native(N, KL, KU, NRHS, (magmaFloat_ptr)APTR, LDDAB, IPIV, (magmaFloat_ptr)BPTR, LDDB, &INFO);
126 magma_sgetmatrix(N, NRHS, (magmaFloat_ptr)BPTR, LDDB, residual.memptr(), LDB, queue);
127
128 return INFO;
129 });
130 }
131
132 if(0 == INFO) {
133 this->pivot.zeros(N);
134 magma_igetmatrix(N, 1, IPIV, N, this->pivot.memptr(), N, queue);
135 }
136 else suanpan_error("Error code {} received, the matrix is probably singular.\n", INFO);
137
138 release();
139
140 return INFO;
141}
