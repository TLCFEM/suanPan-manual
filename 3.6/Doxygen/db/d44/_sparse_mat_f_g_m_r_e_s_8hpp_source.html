67template<sp_d T> int SparseMatBaseFGMRES<T>::direct_solve(Mat<T>& X, const Mat<T>& B) {
68 const auto N = static_cast<int>(B.n_rows);
69
70 const auto restart = std::min(150, N);
71
72 work.zeros((2 * restart + 1) * N + restart * (restart + 9) / 2 + 1);
73
74 X = B;
75
76 csr_form<T, int> csr_mat(this->triplet_mat);
77
78 int request;
79
80 for(auto I = 0llu; I < B.n_cols; ++I) {
81 dfgmres_init(&N, (double*)X.colptr(I), (double*)B.colptr(I), &request, ipar.memptr(), dpar.memptr(), work.memptr());
82 if(request != 0) return request;
83
84 ipar[8] = 1;
85 ipar[9] = 0;
86 ipar[11] = 1;
87 dpar[0] = this->setting.tolerance;
88
89 dfgmres_check(&N, (double*)X.colptr(I), (double*)B.colptr(I), &request, ipar.memptr(), dpar.memptr(), work.memptr());
90 if(request == -1100) return request;
91
92 while(true) {
93 dfgmres(&N, (double*)X.colptr(I), (double*)B.colptr(I), &request, ipar.memptr(), dpar.memptr(), work.memptr());
94 if(request == 0) {
95 int counter;
96 dfgmres_get(&N, (double*)X.colptr(I), (double*)B.colptr(I), &request, ipar.memptr(), dpar.memptr(), work.memptr(), &counter);
97 if(request != 0) return request;
98 suanpan_debug("Iteration counter: {}.\n", counter);
99 break;
100 }
101 if(request != 1) return request;
102 const vec xn(&work[ipar[21] - 1llu], X.n_rows);
103 // ReSharper disable once CppInitializedValueIsAlwaysRewritten
104 // ReSharper disable once CppEntityAssignedButNoRead
105 vec yn(&work[ipar[22] - 1llu], X.n_rows, false, true);
106 yn = csr_mat * xn;
107 }
108 }
109
110 return 0;
111}
