34template<sp_d T> uword round_up(const uword in_size) {

95 void scale_accu(const T scalar, const shared_ptr<MetaMat<T>>& M) override {
96 if(nullptr == M) return;

98 if(this->n_rows != M->n_rows || this->n_cols != M->n_cols || this->n_elem != M->n_elem) throw invalid_argument("size mismatch");

107 if(this->n_rows != M.n_rows || this->n_cols != M.n_cols) throw invalid_argument("size mismatch");
108 this->factored = false;
109 const auto row = M.row_mem();

111 const auto val = M.val_mem();
112 if(1. == scalar) for(auto I = 0llu; I < M.n_elem; ++I) this->at(row[I], col[I]) += val[I];
113 else if(-1. == scalar) for(auto I = 0llu; I < M.n_elem; ++I) this->at(row[I], col[I]) -= val[I];
114 else for(auto I = 0llu; I < M.n_elem; ++I) this->at(row[I], col[I]) += scalar * val[I];
