37 template<typename T> auto sort_color_metis(suanpan::graph<T>& element_register, const int num_color, const char method) {
38  wall_clock timer;
39  timer.tic();
40 
41  const auto element_size = element_register.size();
42 
43  std::atomic num_edges = 0llu;
44  suanpan::for_all(element_register, [&](const suanpan::set<T>& element) { num_edges += element.size(); });
45 
46  std::vector<idx_t> xadj;
47  xadj.reserve(element_size + 1llu);
48  xadj.emplace_back(0);
49  std::vector<idx_t> adjncy;
50  adjncy.reserve(num_edges);
51 
52  for(const auto& element : element_register) {
53  adjncy.insert(adjncy.end(), element.begin(), element.end());
54  xadj.emplace_back(xadj.back() + static_cast<idx_t>(element.size()));
55  }
56 
57  auto nvtxs = static_cast<idx_t>(element_size);
58  idx_t ncon = 1;
59  auto nparts = static_cast<idx_t>(num_color);
60  idx_t edgecut = 0;
61 
62  std::vector<int> part(nvtxs);
63 
64  idx_t* vsize = nullptr;
65  real_t* tpwgts = nullptr;
66  real_t* ubvec = nullptr;
67 
68  idx_t options[METIS_NOPTIONS];
69  METIS_SetDefaultOptions(options);
70 #ifdef SUANPAN_DEBUG
71  options[METIS_OPTION_DBGLVL] = METIS_DBG_INFO | METIS_DBG_TIME;
72 #endif
73  options[METIS_OPTION_OBJTYPE] = METIS_OBJTYPE_VOL;
74 
75  if('K' == method) METIS_PartGraphKway(&nvtxs, &ncon, xadj.data(), adjncy.data(), nullptr, vsize, nullptr, &nparts, tpwgts, ubvec, options, &edgecut, part.data());
76  else METIS_PartGraphRecursive(&nvtxs, &ncon, xadj.data(), adjncy.data(), nullptr, vsize, nullptr, &nparts, tpwgts, ubvec, options, &edgecut, part.data());
77 
78  suanpan_debug("Coloring algorithm takes {:.5E} seconds.\n", timer.toc());
79 
80  return part;
81 }

83 template<typename T> std::vector<std::vector<T>> sort_color_wp(const suanpan::graph<T>& node_register) {
84  wall_clock timer;
85  timer.tic();
86 
87  const auto num_node = node_register.size();
88 
89  uvec weight(num_node, fill::none);
90 
91  suanpan::for_each(node_register.size(), [&](const size_t I) { weight(I) = node_register[I].size(); });
92 
93  auto comparator = [&](const T A, const T B) { return weight[A] > weight[B]; };
94 
95  std::multiset<T, decltype(comparator)> degree(comparator);
96  for(T I = 0; I < num_node; ++I) degree.insert(I);
97 
98  std::vector<std::vector<T>> color_map;
99 
100  while(!degree.empty()) {
101  color_map.emplace_back();
102  auto& color = color_map.back();
103 
104  color.emplace_back(*degree.begin());
105  for(auto node = degree.erase(degree.begin()); node != degree.end();) {
106  auto flag = false;
107  for(const auto colored : color) {
108  if(colored == *node) continue;
109  if(node_register[colored].contains(*node)) {
110  flag = true;
111  break;
112  }
113  }
114  if(flag) ++node;
115  else {
116  color.emplace_back(*node);
117  node = degree.erase(node);
118  }
119  }
120  }
121 
122  for(auto& color : color_map) color.shrink_to_fit();
123  color_map.shrink_to_fit();
124 
125  suanpan_debug("Coloring algorithm takes {:.5E} seconds.\n", timer.toc());
126 
127  return color_map;
128 }

130 template<typename T> std::vector<std::vector<T>> sort_color_mis(const suanpan::graph<T>& node_register) {
131  wall_clock timer;
132  timer.tic();
133 
134  const auto num_node = node_register.size();
135 
136  uvec weight(num_node, fill::none);
137 
138  suanpan::for_each(node_register.size(), [&](const size_t I) { weight(I) = node_register[I].size(); });
139 
140  uword counter = num_node;
141  for(const auto I : sort_index(weight, "descend").eval()) weight[I] = --counter;
142 
143  auto comparator = [&](const T A, const T B) { return weight[A] > weight[B]; };
144 
145  std::multiset<T, decltype(comparator)> degree(comparator);
146  for(T I = 0; I < num_node; ++I) degree.insert(I);
147 
148  std::vector<std::vector<T>> color_map;
149 
150  while(!degree.empty()) {
151  color_map.emplace_back();
152  auto& color = color_map.back();
153 
154  auto degree_copy = degree;
155  do {
156  const auto node_to_color = *degree_copy.begin();
157  color.emplace_back(node_to_color);
158  degree.erase(node_to_color);
159  for(auto neighbour : node_register[node_to_color]) degree_copy.erase(neighbour);
160  }
161  while(!degree_copy.empty());
162  }
163 
164  for(auto& color : color_map) color.shrink_to_fit();
165  color_map.shrink_to_fit();
166 
167  suanpan_debug("Coloring algorithm takes {:.5E} seconds.\n", timer.toc());

169  return color_map;
