349template<sp_d T> int SparseMatMAGMA<T>::direct_solve(Mat<T>& X, Mat<T>&& B) {
350 auto b_rows = static_cast<magma_index_t>(B.n_rows), b_cols = static_cast<magma_index_t>(B.n_cols);
351
352 if(!this->factored) {
353 this->factored = true;
354 csr_mat = csr_form<T, magma_index_t>(this->triplet_mat, SparseBase::ZERO);
355 magma_s::csrset(csr_mat.n_rows, csr_mat.n_cols, csr_mat.row_mem(), csr_mat.col_mem(), csr_mat.val_mem(), &A_host, queue);
356 magma_s::mtransfer(A_host, &A_device, Magma_CPU, Magma_DEV, queue);
357 }
358
359 suanpan::detail::magma::parse_opts(opts, std::istringstream(this->setting.option));
360
361 magma_s::vset(b_rows, b_cols, (T*)B.memptr(), &b_host, queue);
362 magma_s::mtransfer(b_host, &b_device, Magma_CPU, Magma_DEV, queue);
363
364 magma_s::precondsetup(A_device, b_device, &opts.solver_par, &opts.precond_par, queue);
365
366 magma_s::vinit(&x_device, Magma_DEV, b_rows, b_cols, T(0), queue);
367
368 magma_s::solver(A_device, b_device, &x_device, &opts, queue);
369
370 magma_s::mtransfer(x_device, &x_host, Magma_DEV, Magma_CPU, queue);
371
372 X = std::move(B);
373 magma_s::vcopy(x_host, &b_rows, &b_cols, X.memptr(), queue);
374
375 return SUANPAN_SUCCESS;
376}
