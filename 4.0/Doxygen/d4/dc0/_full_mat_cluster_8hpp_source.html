110template<sp_d T, typename solver_t> int FullMatBaseCluster<T, solver_t>::direct_solve(Mat<T>& X, Mat<T>&& B) {
111 if(this->factored) return this->solve_trs(X, std::move(B));
112
113 suanpan_assert([&] { if(this->n_rows != this->n_cols) throw std::invalid_argument("requires a square matrix"); });
114
115 this->factored = true;
116
117 const auto INFO = bcast_from_root(solver.solve({this->n_rows, this->n_cols, this->memptr()}, {B.n_rows, B.n_cols, B.memptr()}));
118
119 if(0 == INFO) bcast_from_root(X = std::move(B));
120 else suanpan_error("Error code {} received, the matrix is probably singular.\n", INFO);
121
122 return INFO;
123}
