64template<sp_d T> int SparseMatBaseMUMPS<T>::alloc() {
65 if(this->factored) return 0;
66
67 dealloc();
68
69 this->factored = true;
70
71 this->triplet_mat.csc_condense();
72
73 mumps_job.job = -1;
74 dmumps_c(&mumps_job);
75
76 mumps_job.n = static_cast<int>(this->triplet_mat.n_rows);
77 mumps_job.nnz = static_cast<int64_t>(this->triplet_mat.n_elem);
78
79 l_irn.set_size(mumps_job.nnz);
80 l_jrn.set_size(mumps_job.nnz);
81
82 suanpan_for(0, static_cast<int>(mumps_job.nnz), [&](const int I) {
83 l_irn[I] = static_cast<int>(this->triplet_mat.row(I) + 1);
84 l_jrn[I] = static_cast<int>(this->triplet_mat.col(I) + 1);
85 });
86
87 mumps_job.irn = l_irn.memptr();
88 mumps_job.jcn = l_jrn.memptr();
89 mumps_job.a = this->triplet_mat.val_mem();
90
91 mumps_job.icntl[0] = -1;
92 mumps_job.icntl[1] = -1;
93 mumps_job.icntl[2] = -1;
94 mumps_job.icntl[3] = 0;
95 mumps_job.icntl[13] = 100;
96 mumps_job.icntl[19] = 0; // dense rhs
97 mumps_job.icntl[32] = 1; // determinant
98 mumps_job.icntl[34] = 1; // BLR
99
100 mumps_job.job = 4;
101 dmumps_c(&mumps_job);
102
103 if(0 != mumps_job.info[0])
104 suanpan_error("Error code {} received.\n", mumps_job.info[0]);
105
106 return mumps_job.info[0];
107}
