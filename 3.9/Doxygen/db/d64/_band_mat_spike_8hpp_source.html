141 Mat<T> Y(arma::size(X));
142
143 const auto M = static_cast<blas_int>(this->n_rows);
144 const auto N = static_cast<blas_int>(this->n_cols);
145 const auto KL = static_cast<blas_int>(l_band);
146 const auto KU = static_cast<blas_int>(u_band);
147 const auto LDA = static_cast<blas_int>(m_rows);
148 constexpr blas_int INC = 1;
149 T ALPHA = T(1);
150 T BETA = T(0);
151
152 if constexpr(std::is_same_v<T, float>) {
153 using E = float;
154 suanpan::for_each(X.n_cols, [&](const uword I) { arma_fortran(arma_sgbmv)(&TRAN, &M, &N, &KL, &KU, (E*)&ALPHA, (E*)this->memptr(), &LDA, (E*)X.colptr(I), &INC, (E*)&BETA, (E*)Y.colptr(I), &INC); });
155 }
156 else {
157 using E = double;
158 suanpan::for_each(X.n_cols, [&](const uword I) { arma_fortran(arma_dgbmv)(&TRAN, &M, &N, &KL, &KU, (E*)&ALPHA, (E*)this->memptr(), &LDA, (E*)X.colptr(I), &INC, (E*)&BETA, (E*)Y.colptr(I), &INC); });

164template<sp_d T> int BandMatSpike<T>::direct_solve(Mat<T>& X, Mat<T>&& B) {
165 if(!this->factored) {
166 suanpan_assert([&] { if(this->n_rows != this->n_cols) throw std::invalid_argument("requires a square matrix"); });
167
168 la_it INFO = 0;
169
170 auto N = static_cast<la_it>(this->n_rows);
171 auto KL = static_cast<la_it>(l_band);
172 auto KU = static_cast<la_it>(u_band);
173 auto LDAB = static_cast<la_it>(m_rows);
174 const auto KLU = std::max(l_band, u_band);
175 this->factored = true;
176
177 if constexpr(std::is_same_v<T, float>) {
178 using E = float;
179 WORK.zeros(KLU * KLU * SPIKE[9]);
180 sspike_gbtrf_(SPIKE, &N, &KL, &KU, (E*)this->memptr(), &LDAB, (E*)WORK.memptr(), &INFO);
181 }
182 else if(Precision::FULL == this->setting.precision) {
183 using E = double;
184 WORK.zeros(KLU * KLU * SPIKE[9]);
185 dspike_gbtrf_(SPIKE, &N, &KL, &KU, (E*)this->memptr(), &LDAB, (E*)WORK.memptr(), &INFO);
186 }
187 else {
188 this->s_memory = this->to_float();
189 SWORK.zeros(KLU * KLU * SPIKE[9]);
190 sspike_gbtrf_(SPIKE, &N, &KL, &KU, this->s_memory.memptr(), &LDAB, SWORK.memptr(), &INFO);
191 }
192
193 if(0 != INFO) {
194 suanpan_error("Error code {} received, the matrix is probably singular.\n", INFO);
195 return INFO;
196 }
197 }
198
199 return this->solve_trs(X, std::move(B));
200}
